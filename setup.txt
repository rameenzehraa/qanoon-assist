------------------------------------------------------
database creation in postgresql
------------------------------------------------------
CREATE DATABASE qanoon_assist;
CREATE USER qanoon_user WITH PASSWORD 'your_secure_password';
ALTER ROLE qanoon_user SET client_encoding TO 'utf8';
ALTER ROLE qanoon_user SET default_transaction_isolation TO 'read committed';
ALTER ROLE qanoon_user SET timezone TO 'Asia/Karachi';
GRANT ALL PRIVILEGES ON DATABASE qanoon_assist TO qanoon_user;
