from users.models import LawyerSpecialty

# List of legal specialties for Pakistan
specialties = [
    'Family Law',
    'Criminal Law', 
    'Corporate Law',
    'Property Law',
    'Civil Law',
    'Tax Law',
    'Immigration Law',
    'Labor Law',
    'Constitutional Law',
    'Banking Law',
    'Intellectual Property Law',
    'Cyber Law',
    'Environmental Law',
    'Human Rights Law',
]

print("Creating lawyer specialties...")

for name in specialties:
    # get_or_create prevents duplicates
    specialty, created = LawyerSpecialty.objects.get_or_create(name=name)
    if created:
        print(f"✓ Created: {name}")
    else:
        print(f"✓ Already exists: {name}")

print(f"\n✅ Successfully created {len(specialties)} lawyer specialties!")
print("You can now exit the shell with: exit()")
